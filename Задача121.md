## Решение на JavaScript

```js
const table = word => {
let map = []
word.split('').map(c => map[c] ? map[c]++ : map[c] = 1)
let keys = Object.keys(map).sort((a,b) => map[b] - map[a])
let res = ""
for(let c in keys) res += (keys[c].repeat(map[keys[c]]))
return res
}

console.log(table("барракуда"))
```
